'use client';
import { useScroll, useTransform, motion } from 'framer-motion';

export default function Home() {
  const { scrollYProgress } = useScroll();
  
  // Levels 0-2: Chaos (Blurry, high noise)
  // Levels 9-11: Logic (Sharp, clean)
  const blur = useTransform(scrollYProgress, [0, 0.5, 1], ['4px', '1px', '0px']);
  const opacity = useTransform(scrollYProgress, [0, 0.2, 0.5], [0.3, 0.6, 1]);

  return (
    <motion.main style={{ filter: `blur(${blur})`, opacity }} className="bg-black text-white">
      {/* Your Sections Map from "my story.md" */}
      <section id="origin" className="h-screen flex items-center px-10">
         <h1 className="text-8xl font-black opacity-20">2007: THE SEED</h1>
      </section>
      
      {/* ... other sections ... */}

      <section id="dashboard" className="min-h-screen p-10">
         <h2 className="text-2xl font-mono text-blue-500 mb-10 tracking-widest">CORE_KERNEL_V1.5</h2>
         <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-4">
            {projects.map(p => <ProjectModule key={p.id} project={p} />)}
         </div>
      </section>
    </motion.main>
  );
}
